import java.util.Scanner;

public class Exercicio_13 {
    public static void main(String[] args) {
        Scanner Calculadora = new Scanner(System.in);
        
        System.out.println("Criar uma calculadora de operações basicas");
        System.out.println("Soma, Subtração, Multiplicação e divisão");
        System.out.println("o algoritimo deve ler dois numeros e o sinal correspondente a operação desejada");
        System.out.println("no final deve ser impresso o resultado");
        System.out.println("______________________________________");
        System.out.println("");
        
        System.out.print("Escolha a operação (+, -, *, /): ");
        char operacao = Calculadora.next().charAt(0);
       
        System.out.print("Digite o primeiro número: ");
        double num1 = Calculadora.nextDouble();

        System.out.print("Digite o segundo número: ");
        double num2 = Calculadora.nextDouble();

       
        double resultado;

     
        switch (operacao) {
            case '+':
                resultado = num1 + num2;
                System.out.printf("Resultado: "+resultado);
                break;
            case '-':
                resultado = num1 - num2;
                System.out.printf("Resultado: "+resultado);
                break;
            case '*':
                resultado = num1 * num2;
                System.out.printf("Resultado: "+resultado);
                break;
            case '/':
                
                if (num2 <= 0) {
                    resultado = num1 / num2;
                    System.out.printf("Resultado: "+resultado);
                } else {
                    System.out.println("Erro: Impossivel Dividir!.");
                }
                break;
            default:
                System.out.println("Operação inválida. Use +, -, * ou /.");
                break;
        }

       
        Calculadora.close();
    }
}
